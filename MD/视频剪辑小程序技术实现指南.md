class MediaManager {
    constructor() {
        this.loadedMedia = new Map();
    }
    
    /**
     * åŠ è½½è§†é¢‘æ–‡ä»¶
     */
    async loadVideo(file) {
        return new Promise((resolve, reject) => {
            const video = document.createElement('video');
            const url = URL.createObjectURL(file);
            
            video.onloadedmetadata = () => {
                const videoData = {
                    id: Date.now(),
                    name: file.name,
                    element: video,
                    duration: video.duration,
                    width: video.videoWidth,
                    height: video.videoHeight,
                    size: file.size,
                    type: file.type,
                    url: url
                };
                
                this.loadedMedia.set(videoData.id, videoData);
                resolve(videoData);
            };
            
            video.onerror = () => {
                URL.revokeObjectURL(url);
                reject(new Error('Failed to load video file'));
            };
            
            video.src = url;
            video.load();
        });
    }
    
    /**
     * åŠ è½½éŸ³é¢‘æ–‡ä»¶
     */
    async loadAudio(file) {
        const audioContext = new (window.AudioContext || window.webkitAudioContext)();
        
        try {
            const arrayBuffer = await file.arrayBuffer();
            const audioBuffer = await audioContext.decodeAudioData(arrayBuffer);
            
            const audioData = {
                id: Date.now(),
                name: file.name,
                buffer: audioBuffer,
                duration: audioBuffer.duration,
                sampleRate: audioBuffer.sampleRate,
                channels: audioBuffer.numberOfChannels,
                size: file.size,
                type: file.type
            };
            
            this.loadedMedia.set(audioData.id, audioData);
            return audioData;
            
        } catch (error) {
            throw new Error('Failed to load audio file: ' + error.message);
        }
    }
    
    /**
     * è·å–åª’ä½“ä¿¡æ¯
     */
    getMediaInfo(id) {
        return this.loadedMedia.get(id);
    }
    
    /**
     * æ¸…ç†åª’ä½“èµ„æº
     */
    cleanup(id) {
        const media = this.loadedMedia.get(id);
        if (media && media.url) {
            URL.revokeObjectURL(media.url);
        }
        this.loadedMedia.delete(id);
    }

class AudioMixer {
    constructor() {

        this.masterGain = this.audioContext.createGain();
        this.masterGain.connect(this.audioContext.destination);
        
        this.tracks = new Map();

        this.startTime = 0;
        this.pauseTime = 0;
    }
    
    /**
     * æ·»åŠ éŸ³é¢‘è½¨é“
     */
    addTrack(trackData) {
        const track = {
            id: trackData.id,
            buffer: trackData.buffer,
            startTime: trackData.startTime,
            duration: trackData.duration,
            volume: trackData.volume || 1.0,
            source: null,
            gainNode: null
        };
        
        this.tracks.set(track.id, track);
        return track;
    }
    
    /**
     * æ’­æ”¾æ‰€æœ‰è½¨é“
     */
    play(currentTime = 0) {
        if (this.isPlaying) return;
        
        this.isPlaying = true;
        this.startTime = this.audioContext.currentTime - currentTime;
        
        // æ’­æ”¾æ‰€æœ‰è½¨é“
        this.tracks.forEach(track => {
            this.playTrack(track, currentTime);
        });
    }
    
    /**
     * æ’­æ”¾å•ä¸ªè½¨é“
     */
    playTrack(track, currentTime) {
        // è®¡ç®—è½¨é“å¼€å§‹æ—¶é—´
        const trackStartTime = Math.max(0, track.startTime - currentTime);
        const audioStartTime = Math.max(0, currentTime - track.startTime);
        
        // æ£€æŸ¥æ˜¯å¦éœ€è¦æ’­æ”¾
        if (currentTime > track.startTime + track.duration) {
            return; // è½¨é“å·²ç»“æŸ
        }
        
        // åˆ›å»ºéŸ³é¢‘æº
        const source = this.audioContext.createBufferSource();
        const gainNode = this.audioContext.createGain();
        
        source.buffer = track.buffer;
        source.connect(gainNode);
        gainNode.connect(this.masterGain);
        
        // è®¾ç½®éŸ³é‡
        gainNode.gain.value = track.volume;
        
        // å¼€å§‹æ’­æ”¾
        const when = this.audioContext.currentTime + trackStartTime;
        const offset = audioStartTime;
        const duration = track.duration - audioStartTime;
        
        source.start(when, offset, duration);
        
        // ä¿å­˜å¼•ç”¨
        track.source = source;
        track.gainNode = gainNode;
    }
    
    /**
     * æš‚åœæ’­æ”¾
     */
    pause() {
        if (!this.isPlaying) return;
        
        this.isPlaying = false;
        this.pauseTime = this.audioContext.currentTime - this.startTime;
        
        // åœæ­¢æ‰€æœ‰è½¨é“
        this.tracks.forEach(track => {
            if (track.source) {
                track.source.stop();
                track.source = null;
            }
        });
    }
    
    /**
     * åœæ­¢æ’­æ”¾
     */
    stop() {
        this.pause();
        this.pauseTime = 0;
    }
    
    /**
     * è®¾ç½®è½¨é“éŸ³é‡
     */
    setTrackVolume(trackId, volume) {
        const track = this.tracks.get(trackId);
        if (track) {
            track.volume = volume;
            if (track.gainNode) {
                track.gainNode.gain.value = volume;
            }
        }
    }

    setCurrentTime(time) {

    }

            bgY -= bgHeight;
        }

            // è®¾ç½®å¯¼å‡ºç”»å¸ƒ

            // è·å–ç”»å¸ƒæµ

            // æ”¶é›†å½•åˆ¶æ•°æ®

                await new Promise(resolve => {
                    video.onseeked = resolve;
                    if (video.readyState >= 2) resolve();
                });
                
                this.exportCtx.drawImage(video, 0, 0, this.exportCanvas.width, this.exportCanvas.height);
            }
            
            // æ¸²æŸ“æ–‡å­—
            project.tracks.text.forEach(textTrack => {
                if (currentTime >= textTrack.startTime && 
                    currentTime <= textTrack.startTime + textTrack.duration) {
                    const textRenderer = new TextRenderer();
                    textRenderer.render(this.exportCtx, textTrack);
                }
            });
            
            // ç­‰å¾…ä¸‹ä¸€å¸§
            await new Promise(resolve => setTimeout(resolve, 16));
        }
    }
    
    /**
     * åˆ›å»ºéŸ³é¢‘æµ
     */
    async createAudioStream(project) {
        if (project.tracks.audio.length === 0) return null;
        
        try {
            const audioContext = new AudioContext();
            const destination = audioContext.createMediaStreamDestination();
            
            // æ··åˆæ‰€æœ‰éŸ³é¢‘è½¨é“
            project.tracks.audio.forEach(audioTrack => {
                const source = audioContext.createBufferSource();
                const gainNode = audioContext.createGain();
                
                source.buffer = audioTrack.buffer;
                source.connect(gainNode);
                gainNode.connect(destination);
                
                gainNode.gain.value = audioTrack.volume;
                source.start(audioTrack.startTime);
            });
            
            return destination.stream;
            
        } catch (error) {
            console.warn('Failed to create audio stream:', error);
            return null;
        }
    }
    
    /**
     * è·å–æ”¯æŒçš„MIMEç±»å‹
     */
    getSupportedMimeType(format) {
        const types = {
            'webm': ['video/webm;codecs=vp9', 'video/webm;codecs=vp8', 'video/webm'],
            'mp4': ['video/mp4;codecs=h264', 'video/mp4']
        };
        
        const candidates = types[format] || types['webm'];
        
        for (const type of candidates) {
            if (MediaRecorder.isTypeSupported(type)) {
                return type;
            }
        }
        
        return 'video/webm';
    }
    
    /**
     * è·å–æ¯”ç‰¹ç‡
     */
    getBitrate(quality) {
        const bitrates = {
            'low': 2000000,    // 2 Mbps
            'medium': 5000000, // 5 Mbps
            'high': 8000000,   // 8 Mbps
            'ultra': 15000000  // 15 Mbps
        };
        
        return bitrates[quality] || bitrates['high'];
    }
    
    /**
     * ç”Ÿæˆæ–‡ä»¶å
     */
    generateFilename(format) {
        const now = new Date();
        const timestamp = now.toISOString().replace(/[:.]/g, '-').slice(0, -5);
        const extension = format === 'mp4' ? 'mp4' : 'webm';
        return `video-edit-${timestamp}.${extension}`;
    }
    
    /**
     * æ¸…ç†èµ„æº
     */
    cleanup() {
        if (this.exportCanvas) {
            this.exportCanvas = null;
            this.exportCtx = null;
        }
        
        if (this.mediaRecorder) {
            this.mediaRecorder = null;
        }
    }
}
